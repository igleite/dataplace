SELECT  db.[name],db.[recovery_model_desc],db.[log_reuse_wait_desc],ls.[cntr_value] AS [LogSize_KB],lu.[cntr_value] AS [LogUsed_KB],db.[compatibility_level],db.[page_verify_option_desc],  CAST(   CAST(lu.[cntr_value] AS FLOAT) /        CASE WHEN CAST(ls.[cntr_value] AS FLOAT) = 0                THEN 1                ELSE CAST(ls.[cntr_value] AS FLOAT)        END AS DECIMAL(18,2)) * 100 AS [Percente_Log_Used]  FROM [sys].[databases] AS db  JOIN [sys].[dm_os_performance_counters] AS lu  ON db.[name] = lu.[instance_name]  JOIN [sys].[dm_os_performance_counters] AS ls  ON db.[name] = ls.[instance_name]  WHERE   lu.[counter_name] LIKE 'Log File(s) Used Size (KB)%' AND ls.[counter_name] LIKE 'Log File(s) Size (KB)%'  AND name = (SELECT DB_NAME())
