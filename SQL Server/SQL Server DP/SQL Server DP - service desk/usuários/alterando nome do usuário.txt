UPDATE ANUC
SET ANUC.ClaimValue = 'Maria Eduarda dos Santos'
FROM AspNetUsers ANU
         INNER JOIN AspNetUserClaims ANUC ON ANU.Id = ANUC.UserId
WHERE ANU.Email = 'helpdesk05@dataplace.com.br'
  AND ANUC.ClaimType = 'Name'
